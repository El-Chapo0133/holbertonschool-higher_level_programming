#!/usr/bin/python3

def add_integer(a, b):
    """
    >>> add_integer(1, 2)
    3
    >>> add_integer(3, 4)
    7
    """
    if (not isinstance(a, int)):
        raise TypeError("a must be an integer")
    elif (not isinstance(b, int)):
        raise TypeError("b must be an integer")

    return a + b
